dirin = "data/originals/T10us/Ar39_1to7Pileups/"
ar39_groups = []
for grp in range(1, 7+1):
    print("[Info] Loading Group {}".format(grp))
    for i, file in enumerate(glob.glob(os.path.join(dirin, "*Grp{}*".format(grp)))):
        if i==0:
            df = pd.read_csv(file, comment='#', index_col=False)
            df = df[df.columns[:-1]]
        else:
            tmpdf = pd.read_csv(file, comment='#', index_col=False)
            tmpdf = tmpdf[tmpdf.columns[:-1]]
            df = pd.concat([df, tmpdf])
    df = df[df.pedetected > 0]
    ar39_groups.append(df.tail(100000))

import time

start = time.time()
ar39_predictions = []
print("[Info] Starting predicting Ar39")
for i, groupdata in enumerate(ar39_groups):
    print("     Starting Group {} ({} instances)...".format(i+1, len(groupdata)))
    batch_size = 32
    start_group = time.time()
    dfall = groupdata.to_numpy()
    dfall = dfall.reshape((dfall.shape[0], dfall.shape[1], 1))
    dfall = dfall[:, 3:]
    dataset = tf.data.Dataset.from_tensor_slices(dfall).batch(batch_size)
    preds = np.zeros((dfall.shape[0], 1))
    for j, batch in enumerate(dataset):
        preds[j * batch_size : (j + 1) * batch_size] = model(batch)
    ar39_predictions.append(np.array(preds))
    errors = len(np.nonzero(ar39_predictions[i] > 0.5)[0])
    tot = len(ar39_predictions[i])
    print("     Group {}: Error Rate: {:.3f}%, Time: {}".format(i+1, errors/tot*100, time.time() - start_group))
print("[Info] Done. Time: {}".format(time.time() - start))
